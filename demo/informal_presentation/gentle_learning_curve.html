<script>
		if ( ! Detector.webgl ) Detector.addGetWebGLMessage();
		
		var sourceURL = "graphs/introduction.xgml",
		visualizer, graph, layoutEngine, space, controls, loader;
		
		$.ajax({
				type: "GET",
				url: sourceURL,
				dataType: "xml",
				success: function(data) {
				
					loader = new TRIPTYCH.BasicGraphLoader();
					graph = loader.load('xgml', data);
					visualizer = new TRIPTYCH.TextVisualizer(); 
					layoutEngine = new TRIPTYCH.SpringForceLayoutEngine();
					controls = new TRIPTYCH.BasicControls();
					space = new TRIPTYCH.Space(graph, visualizer, layoutEngine, controls);
					space.init();
					animate();
	
				}
			});
		
		function animate() {
	
			requestAnimationFrame( animate );
	
			space.update();
	
		}
		
		</script>
